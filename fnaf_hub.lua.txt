local function saveLearned(tbl)
    if writefile then pcall(writefile, LEARN_FILE, HttpService:JSONEncode(tbl)) end
end
local function loadLearned()
    if isfile and isfile(LEARN_FILE) and readfile then
        local ok, data = pcall(function() return HttpService:JSONDecode(readfile(LEARN_FILE)) end)
        if ok and type(data) == "table" then return data end
    end
    return {}
end

--===================== SAFE FIRE HELPERS =========================--
local function safeFire(remote, ...)
    if not remote then return false end
    local args = {...}
    local ok = pcall(function() remote:FireServer(unpack(args)) end)
    return ok
end

local function notify(title, text, dur)
    pcall(function()
        StarterGui:SetCore("SendNotification", { Title = title, Text = text, Duration = dur or 4 })
    end)
end

--===================== DOOR/LIGHT + LEARNING =====================--
-- learned[remotePath] = { path=..., args={...} }
local Learned = loadLearned()

-- part->control map so we recognise clicks on office buttons
local function controlKeyFromInstance(inst)
    if inst == Office.LeftDoorBtn  or inst == Office.LeftDoor  then return "leftDoor"  end
    if inst == Office.RightDoorBtn or inst == Office.RightDoor then return "rightDoor" end
    if inst == Office.LeftLightBtn  then return "leftLight"  end
    if inst == Office.RightLightBtn then return "rightLight" end
    return nil
end

-- hook __namecall to LEARN the exact fire the game uses on E-press
if hookmetamethod and getnamecallmethod and checkcaller then
    local old
    old = hookmetamethod(game, "__namecall", function(self, ...)
        local method = getnamecallmethod()
        if (method == "FireServer") and not checkcaller() then
            local args = { ... }
            -- direct OfficeControls events
            for key, rem in pairs(Remotes) do
                if self == rem and (key=="leftDoor" or key=="rightDoor" or key=="leftLight" or key=="rightLight") then
                    Learned[key] = { kind="remote", key=key, args=args }
                    saveLearned(Learned)
                end
            end
            -- click-based control (playerUsedClickOnObjectEvent(button))
            if self == Remotes.clickObj and typeof(args[1]) == "Instance" then
                local ck = controlKeyFromInstance(args[1])
                if ck then
                    Learned[ck] = { kind="click", key=ck, target=args[1] }
                    saveLearned(Learned)
                end
            end
        end
        return old(self, ...)
    end)
end

-- toggle a control. Priority order, all input-free (mobile + PC):
--   1) fireclickdetector on the control's ClickDetector  (works everywhere)
--   2) learned replay (exact remote your tap used, if captured)
--   3) direct OfficeControls remote fire
--   4) playerUsedClickOnObjectEvent(button)
local function toggleControl(key)
    -- 1) ClickDetectors -- the mobile-proof path
    if fireclickdetector then
        for _, cd in ipairs(ClickDets[key] or {}) do
            if cd then
                local ok = pcall(fireclickdetector, cd)
                if ok then return true end
            end
        end
    end
    -- 2) learned replay
    local rec = Learned[key]
    if Config.PreferLearned and rec then
        if rec.kind == "remote" then
            return safeFire(Remotes[key], unpack(rec.args or {}))
        elseif rec.kind == "click" and rec.target then
            return safeFire(Remotes.clickObj, rec.target)
        end
    end
    -- 3) direct OfficeControls event (stateful toggle, no args)
    if safeFire(Remotes[key]) then return true end
    -- 4) generic click remote on the button part
    local btn = ({
        leftDoor=Office.LeftDoorBtn, rightDoor=Office.RightDoorBtn,
        leftLight=Office.LeftLightBtn, rightLight=Office.RightLightBtn,
    })[key]
    if btn then return safeFire(Remotes.clickObj, btn) end
    return false
end

-- track intended door state so we don't spam toggles
local DoorState = { left=false, right=false }  -- false = open, true = closed
local function setDoor(side, closed)
    if DoorState[side] == closed then return end
    toggleControl(side == "left" and "leftDoor" or "rightDoor")
    DoorState[side] = closed
end

--===================== CAMERA HELPERS ============================--
local monitorUp = false
local function setMonitor(up)
    if monitorUp == up then return end
    safeFire(Remotes.nightGuard, up)
    monitorUp = up
end
local function switchCam(id) safeFire(Remotes.switchCam, id) end
local function flashCam() safeFire(Remotes.camFlash, true, 1) end

-- peek Pirate Cove briefly to reset Foxy's meter
local peeking = false
local function checkFoxy()
    if peeking then return end
    peeking = true
    task.spawn(function()
        local wasUp = monitorUp
        setMonitor(true)
        task.wait(0.15)
        switchCam(PIRATE_COVE)
        task.wait(Config.FoxyCamPeekTime)
        if not wasUp then setMonitor(false) end
        peeking = false
    end)
end

--===================== POSITION UTILS ============================--
local function rootOf(model)
    if not model then return nil end
    return model:FindFirstChild("HumanoidRootPart")
        or model:FindFirstChildWhichIsA("BasePart")
        or (model:IsA("BasePart") and model or nil)
end

local function myRoot()
    local ch = LocalPlayer.Character
    return ch and (ch:FindFirstChild("HumanoidRootPart") or ch:FindFirstChildWhichIsA("BasePart"))
end

local function anchorFor(side)
    return side == "left" and Office.LeftDoorBtn or Office.RightDoorBtn
end

local function distToDoor(model, side)
    local r = rootOf(model)
    local a = anchorFor(side)
    if not r or not a then return math.huge end
    return (r.Position - a.Position).Magnitude
end

--===================== ESP =======================================--
local espObjects = {}
local function clearESP()
    for _, o in pairs(espObjects) do pcall(function() o:Destroy() end) end
    espObjects = {}
end

local function buildESP()
    clearESP()
    for _, a in ipairs(Animatronics) do
        if a.model and a.model.Parent then
            local hl = Instance.new("Highlight")
            hl.Name = "FNAF_ESP"
            hl.FillColor = Config.EspColor
            hl.OutlineColor = Color3.new(1,1,1)
            hl.FillTransparency = 0.55
            hl.OutlineTransparency = 0
            hl.Adornee = a.model
            hl.Parent = a.model
            espObjects[#espObjects+1] = hl

            local bb = Instance.new("BillboardGui")
            bb.Name = "FNAF_ESP_TAG"
            bb.Size = UDim2.fromOffset(190, 34)
            bb.StudsOffset = Vector3.new(0, 3.2, 0)
            bb.AlwaysOnTop = true
            bb.Adornee = rootOf(a.model)
            bb.Parent = a.model
            local lbl = Instance.new("TextLabel")
            lbl.BackgroundTransparency = 1
            lbl.Size = UDim2.fromScale(1,1)
            lbl.Font = Enum.Font.GothamBold
            lbl.TextSize = 14
            lbl.TextColor3 = Config.EspColor
            lbl.TextStrokeTransparency = 0
            lbl.Text = a.name
            lbl.Parent = bb
            espObjects[#espObjects+1] = bb
            a._tag = lbl
        end
    end
end

RunService.RenderStepped:Connect(function()
    if not State.ESP then return end
    local mr = myRoot()
    for _, a in ipairs(Animatronics) do
        if a._tag and a.model and a.model.Parent then
            if Config.EspShowDistance and mr then
                local r = rootOf(a.model)
                if r then
                    a._tag.Text = ("%s  [%dm]"):format(a.name, math.floor((r.Position-mr.Position).Magnitude))
                end
            else
                a._tag.Text = a.name
            end
        end
    end
end)

--===================== AUTOMATION LOOPS ==========================--
-- proximity auto-door: close the correct side when a threat is near
task.spawn(function()
    while true do
        task.wait(Config.ScanRate)
        if State.AutoDoor then
            local nearL, nearR = false, false
            for _, a in ipairs(Animatronics) do
                if a.side == "left" or a.side == "right" then
                    local d = distToDoor(a.model, a.side)
                    if d <= Config.DoorTriggerDistance then
                        if a.side == "left" then nearL = true else nearR = true end
                    end
                end
            end
            -- close on threat, reopen when clearly gone (power saving)
            if nearL then setDoor("left", true)
            elseif DoorState.left then
                local minL = math.huge
                for _, a in ipairs(Animatronics) do
                    if a.side=="left" then minL = math.min(minL, distToDoor(a.model,"left")) end
                end
                if minL > Config.DoorReopenDistance then setDoor("left", false) end
            end
            if nearR then setDoor("right", true)
            elseif DoorState.right then
                local minR = math.huge
                for _, a in ipairs(Animatronics) do
                    if a.side=="right" then minR = math.min(minR, distToDoor(a.model,"right")) end
                end
                if minR > Config.DoorReopenDistance then setDoor("right", false) end
            end
        end
    end
end)

-- auto-Foxy: timed Pirate Cove peek + emergency left-door slam
task.spawn(function()
    local last = 0
    while true do
        task.wait(0.2)
        if State.AutoFoxy then
            -- emergency: Foxy left the cove -> slam left door now
            local foxy
            for _, a in ipairs(Animatronics) do if a.name=="Foxy" then foxy=a end end
            if foxy and foxy.model then
                local d = distToDoor(foxy.model, "left")
                if d <= (Config.DoorTriggerDistance + 12) then
                    setDoor("left", true)
                end
            end
            if tick() - last >= Config.FoxyCheckInterval then
                last = tick()
                checkFoxy()
            end
        end
    end
end)

-- auto-cam: keep the monitor rotating to stall movement
task.spawn(function()
    local i = 1
    while true do
        task.wait(2)
        if State.AutoCam then
            setMonitor(true)
            switchCam(CAM_LIST[i])
            i = i % #CAM_LIST + 1
        end
    end
end)

-- auto-task: keep firing task completion if the game exposes it
task.spawn(function()
    while true do
        task.wait(3)
        if State.AutoTask and Remotes.updateTask then
            safeFire(Remotes.updateTask)
        end
    end
end)

--===================== AUTO-PLAY NIGHT BOT =======================--
-- Reads the clock text and manages doors/foxy/power until 6AM,
-- then advances to the next night automatically.
local function readNightHour()
    -- TimeScreen shows current hour; parse "12 AM".."5 AM"
    local lbl = resolve(Workspace,"GameTriggers","CameraScreen") -- may vary; scan for a TextLabel named 'time'
    for _, d in ipairs(Workspace:GetDescendants()) do
        if d:IsA("TextLabel") and d.Name == "time" then
            local h = tostring(d.Text):match("%d+")
            if h then return tonumber(h) end
        end
    end
    return nil
end

local autoPlayThread
local function startAutoPlay(allNights)
    if autoPlayThread then return end
    State.AutoPlay = true
    State.AutoDoor = true
    State.AutoFoxy = true
    notify("FNAF Hub", allNights and "Auto-playing ALL nights" or "Auto-playing current night", 5)
    autoPlayThread = task.spawn(function()
        while State.AutoPlay do
            task.wait(1)
            local hr = readNightHour()
            -- keep monitor down by default so doors read correctly; foxy peeks handle cams
            -- (door + foxy loops already run via their flags)
            if hr == 6 then
                notify("FNAF Hub", "Night cleared (6 AM)", 5)
                -- open doors to reset, wait for next-night load
                setDoor("left", false); setDoor("right", false)
                task.wait(6)
                if not allNights then break end
                -- next night loads automatically; reset door state
                DoorState.left, DoorState.right = false, false
            end
        end
        State.AutoPlay = false
        autoPlayThread = nil
    end)
end
local function stopAutoPlay()
    State.AutoPlay = false
    State.AutoAllNights = false
end
