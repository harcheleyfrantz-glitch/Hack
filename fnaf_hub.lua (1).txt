    ------------------------------------------------------------------
      THE DOOR PROBLEM, SOLVED WITHOUT TESTING:
        The office doors (OfficeButtons.LeftDoor / RightDoor) are
        MeshParts that physically move when closed. So the beater
        fires a candidate toggle method, WATCHES the door's position,
        and if it actually moved it locks that method in. It auto-
        learns the real trigger by observation -- no notifications,
        no guessing, no manual tester.

      CONFIRMED FROM DUMPS:
        Cameras : playerSwitchCamsEvent("CAMx")  |  ""  = close tablet
        Foxy    : Pirate Cove = CAM1C (peek to reset him)
        Threats : Bonnie/Foxy = LEFT door, Chica/Freddy = RIGHT door

local function saveJSON(p,t) if writefile then pcall(writefile,p,Http:JSONEncode(t)) end end
local function loadJSON(p)
    if isfile and isfile(p) and readfile then
        local ok,d=pcall(function() return Http:JSONDecode(readfile(p)) end)
        if ok and type(d)=="table" then return d end
    end
    return {}
end

--========================== HELPERS ===============================--
local function fire(rem,...) if rem then return (pcall(function() rem:FireServer(...) end)) end return false end
local function fcd(cd) if cd and fireclickdetector then pcall(fireclickdetector, cd, 50) end end

local function rootOf(m)
    if not m then return nil end
    return m:FindFirstChild("HumanoidRootPart") or m:FindFirstChildWhichIsA("BasePart") or (m:IsA("BasePart") and m or nil)
end
local function myRoot()
    local c=LP.Character
    return c and (c:FindFirstChild("HumanoidRootPart") or c:FindFirstChildWhichIsA("BasePart"))
end
local function partPos(p) return p and p:IsA("BasePart") and p.Position or (p and rootOf(p) and rootOf(p).Position) end

--==================== SELF-LEARNING DOOR CTRL ====================--
-- Candidate toggle methods per side. We fire one, watch the door
-- mesh position; whichever moves it wins and is cached.
local learnedMethod = loadJSON(LEARN)   -- { left=<n>, right=<n> }

local function candidate(side, i)
    local d = Doors[side]
    if     i==1 then fire(Remotes.clickObj, d.btnPart)
    elseif i==2 then fire(Remotes.clickObj, d.doorMesh)
    elseif i==3 then fire(Remotes.clickObj, d.btnModel)
    elseif i==4 then fcd(d.cdBtn)
    elseif i==5 then fcd(d.cdDoor)
    elseif i==6 then fire(d.ev)
    elseif i==7 then fire(d.ev, true)
    elseif i==8 then fire(Remotes.openClose, side)
    end
end
local N_CANDIDATES = 8

-- open-position baseline (doors start open)
local openPos = {}
task.defer(function()
    task.wait(1)
    for s,d in pairs(Doors) do openPos[s] = d.doorMesh and d.doorMesh.Position end
end)

local function doorMoved(side, before)
    local now = Doors[side].doorMesh and Doors[side].doorMesh.Position
    if not now or not before then return false end
    return (now - before).Magnitude > 1.5
end
local function isClosed(side)
    local base, now = openPos[side], Doors[side].doorMesh and Doors[side].doorMesh.Position
    if not base or not now then return false end
    return (now - base).Magnitude > 1.5
end

-- fire the (learned or discovered) toggle once, confirmed by movement
local function toggleDoor(side)
    local d = Doors[side]
    if not d.doorMesh then return false end
    if learnedMethod[side] then
        local before = d.doorMesh.Position
        candidate(side, learnedMethod[side])
        return true
    end
    -- discover: try candidates until the door physically moves
    for i = 1, N_CANDIDATES do
        local before = d.doorMesh.Position
        candidate(side, i)
        local t0 = tick()
        repeat RunService.Heartbeat:Wait() until tick()-t0 > 0.5
        if doorMoved(side, before) then
            learnedMethod[side] = i
            saveJSON(LEARN, learnedMethod)
            return true
        end
    end
    return false
end

-- desired state driver
local function setDoor(side, closed)
    if isClosed(side) == closed then return end
    toggleDoor(side)
end

-- light toggle (non-critical, best-effort)
local function toggleLight(side)
    local l = Lights[side]; if not l then return end
    fire(Remotes.clickObj, l.part); fcd(l.cd); fire(l.ev)
end

--========================== CAMERA ===============================--
local monitorUp, lastCam = false, "CAM1A"
local function switchCam(id) fire(Remotes.switchCam, id); if id=="" then monitorUp=false else monitorUp=true; lastCam=id end end
local function openMonitor(id) switchCam(id or lastCam) end
local function closeMonitor() switchCam("") end
local function flashCam() fire(Remotes.camFlash, true, 1) end

local peeking=false
local function peekFoxy()
    if peeking then return end
    peeking=true
    task.spawn(function()
        local wasUp=monitorUp
        switchCam(PIRATE_COVE); task.wait(0.35)
        if not wasUp then closeMonitor() end
        peeking=false
    end)
end

--========================== DISTANCES ============================--
local function anchor(side) return Doors[side].doorMesh or Doors[side].btnPart end
local function distToDoor(m, side)
    local r=rootOf(m); local a=anchor(side)
    if not r or not a then return math.huge end
    return (r.Position - a.Position).Magnitude
end

local Cfg = {
    trigger = 30, reopen = 45, foxyInt = 5, scan = 0.15,
    espColor = Color3.fromRGB(255,40,40), espDist = true,
}

--========================== ESP ==================================--
local espO={}
local function clearESP() for _,o in pairs(espO) do pcall(function() o:Destroy() end) end espO={} end
local function buildESP()
    clearESP()
    for _,a in ipairs(Animatronics) do
        if a.model and a.model.Parent then
            local hl=Instance.new("Highlight")
            hl.FillColor=Cfg.espColor; hl.OutlineColor=Color3.new(1,1,1); hl.FillTransparency=0.55
            hl.Adornee=a.model; hl.Parent=a.model; espO[#espO+1]=hl
            local bb=Instance.new("BillboardGui")
            bb.Size=UDim2.fromOffset(190,34); bb.StudsOffset=Vector3.new(0,3.2,0); bb.AlwaysOnTop=true
            bb.Adornee=rootOf(a.model); bb.Parent=a.model
            local t=Instance.new("TextLabel")
            t.BackgroundTransparency=1; t.Size=UDim2.fromScale(1,1); t.Font=Enum.Font.GothamBold
            t.TextSize=14; t.TextColor3=Cfg.espColor; t.TextStrokeTransparency=0; t.Text=a.name; t.Parent=bb
            espO[#espO+1]=bb; a._tag=t
        end
    end
end
RunService.RenderStepped:Connect(function()
    if not S.esp then return end
    local mr=myRoot()
    for _,a in ipairs(Animatronics) do
        if a._tag and a.model and a.model.Parent then
            local r=rootOf(a.model)
            if Cfg.espDist and mr and r then a._tag.Text=("%s [%dm]"):format(a.name, math.floor((r.Position-mr.Position).Magnitude))
            else a._tag.Text=a.name end
        end
    end
end)

--==================== CORE DEFENSE LOGIC =========================--
-- close the side a threat is approaching; reopen when clear (power)
local function defendTick()
    local nearL,nearR=false,false
    for _,a in ipairs(Animatronics) do
        if a.side=="left" or a.side=="right" then
            if distToDoor(a.model,a.side) <= Cfg.trigger then
                if a.side=="left" then nearL=true else nearR=true end
            end
        end
    end
    if nearL then setDoor("left",true)
    elseif isClosed("left") then
        local mn=math.huge
        for _,a in ipairs(Animatronics) do if a.side=="left" then mn=math.min(mn,distToDoor(a.model,"left")) end end
        if mn>Cfg.reopen then setDoor("left",false) end
    end
    if nearR then setDoor("right",true)
    elseif isClosed("right") then
        local mn=math.huge
        for _,a in ipairs(Animatronics) do if a.side=="right" then mn=math.min(mn,distToDoor(a.model,"right")) end end
        if mn>Cfg.reopen then setDoor("right",false) end
    end
end

-- proximity loop
task.spawn(function()
    while true do
        task.wait(Cfg.scan)
        if S.autoDoor or S.beat then pcall(defendTick) end
    end
end)
-- foxy loop
task.spawn(function()
    local last=0
    while true do
        task.wait(0.2)
        if S.autoFoxy or S.beat then
            local foxy for _,a in ipairs(Animatronics) do if a.name=="Foxy" then foxy=a end end
            if foxy and foxy.model and distToDoor(foxy.model,"left") <= (Cfg.trigger+14) then setDoor("left",true) end
            if tick()-last >= Cfg.foxyInt then last=tick(); peekFoxy() end
        end
    end
end)
-- cam cycle
task.spawn(function()
    local i=1
    while true do task.wait(2); if S.autoCam then switchCam(CAM_LIST[i]); i=i%#CAM_LIST+1 end end
end)
-- tasks
task.spawn(function()
    while true do task.wait(3); if (S.autoTask or S.beat) and Remotes.updateTask then fire(Remotes.updateTask) end end
end)

--========================== NIGHT HOUR ===========================--
local function nightHour()
    for _,d in ipairs(Workspace:GetDescendants()) do
        if d:IsA("TextLabel") and d.Name=="time" then
            local h=tostring(d.Text):match("%d+"); if h then return tonumber(h) end
        end
    end
end

--========================== AUTO BEATER ==========================--
local beatThread
local function startBeat(allNights, notifyFn)
    if beatThread then return end
    S.beat=true
    if notifyFn then notifyFn(allNights and "Beating ALL nights (1-6)" or "Beating current night") end
    beatThread=task.spawn(function()
        while S.beat do
            task.wait(1)
            local hr=nightHour()
            if hr==6 then
                if notifyFn then notifyFn("Night cleared - 6 AM") end
                setDoor("left",false); setDoor("right",false)
                task.wait(6)
                if not allNights then break end
            end
        end
        S.beat=false; beatThread=nil
    end)
end
local function stopBeat() S.beat=false end
